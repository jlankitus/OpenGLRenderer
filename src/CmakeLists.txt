cmake_minimum_required(VERSION 3.10)

# Set project name
project(OpenGLRenderer)

# Set C++ standard
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add executable
add_executable(OpenGLRenderer src/main.cpp)

# Include directories
include_directories(/usr/include /usr/include/GLFW /usr/include/GL /usr/include/glew /usr/include/glm)

# Link libraries
target_link_libraries(OpenGLRenderer glfw GLEW GL)
